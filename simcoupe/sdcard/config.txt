# circle-coupe: SimCoupe SAM Coupe emulator
# Single config.txt for all supported Raspberry Pi models.
# The firmware automatically applies only the section matching the board.

# ---- Common (all models) ----
arm_64bit=0
hdmi_force_hotplug=1
hdmi_group=2
hdmi_mode=9
disable_overscan=1
disable_splash=1
dtparam=audio=on
enable_uart=1

# ---- Raspberry Pi 2B ----
[pi2]
kernel=kernel7.img
gpu_mem=64
screen_width=800
screen_height=600
# Pi2 has no Bluetooth — PL011 UART available directly, no overlay needed
# Audio configuration - PWM works reliably on Pi 2
sounddev=sndpwm

# ---- Raspberry Pi 3B ----
[pi3]
kernel=kernel8-32.img
gpu_mem=128
screen_width=800
screen_height=600
# Disable Bluetooth to free PL011 UART for serial debug on GPIO 14/15
dtoverlay=disable-bt
# Audio configuration - PWM works on all setups, HDMI requires compatible monitor
# sounddev=sndpwm          # PWM audio (headphone jack) - default, always works
# sounddev=sndhdmi         # HDMI audio (requires HDMI monitor with audio support)
sounddev=sndpwm

# ---- Raspberry Pi 4B / 400 ----
[pi4]
kernel=kernel7l.img
gpu_mem=128
screen_width=800
screen_height=600
# Disable Bluetooth to free PL011 UART for serial debug on GPIO 14/15
dtoverlay=disable-bt
# Pi 4/400 specific: Circle kernel7l.img uses the same dtb as pi3
# Force bcm2711-rpi-4-b.dtb for Pi 4B (same as pi4b-p400 in firmware list)
# Note: Pi 400 uses bcm2711-rpi-400.dtb, firmware auto-selects it
# Audio configuration - PWM works on all setups, HDMI requires compatible monitor
# sounddev=sndpwm          # PWM audio (headphone jack) - default, always works
# sounddev=sndhdmi         # HDMI audio (requires HDMI monitor with audio support)
sounddev=sndpwm
max_framebuffers=2

# ---- Back to all models ----
[all]
