# circle-coupe: SimCoupe SAM Coupe emulator on Raspberry Pi 3B
# Place this file on the FAT32 boot partition of the SD card.

# ---- Boot ----
# 32-bit kernel (AArch32)
arm_64bit=0
# Kernel filename for RPi3 (ARMv8 in 32-bit mode)
kernel=kernel8-32.img

# ---- Video ----
# Force HDMI output even if no display detected at boot
hdmi_force_hotplug=1
# 800x600 @ 60Hz (DMT mode 9) — standard resolution, no overscan
hdmi_group=2
hdmi_mode=9
# Disable overscan
disable_overscan=1

# ---- Serial debug ----
# Disable Bluetooth (frees PL011 UART for serial on GPIO 14/15)
dtoverlay=disable-bt
# Enable UART (115200 baud on GPIO 14 TX / 15 RX)
enable_uart=1

# ---- GPU/Memory ----
# 1280*720*4*2 (double-buffer) = ~7.4MB framebuffer
gpu_mem=128

# ---- Circle screen parameters ----
# Required for CKernelOptions::GetWidth/Height()
screen_width=1280
screen_height=720

# ---- Audio ----
dtparam=audio=on
# Force audio output to HDMI (0=auto, 1=jack, 2=HDMI)
hdmi_drive=2

# ---- Misc ----
# Disable rainbow splash screen for faster boot
disable_splash=1
# No overscan (use full HDMI area)
disable_overscan=1
