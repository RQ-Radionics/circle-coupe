# circle-coupe: SimCoupe SAM Coupe emulator
# Single config.txt for all supported Raspberry Pi models.
# The firmware automatically applies only the section matching the board.

# ---- Common (all models) ----
arm_64bit=0
hdmi_force_hotplug=1
hdmi_group=2
hdmi_mode=9
disable_overscan=1
disable_splash=1
dtparam=audio=on
enable_uart=1

# ---- Raspberry Pi 2B ----
[pi2]
kernel=kernel7.img
gpu_mem=64
screen_width=800
screen_height=600
# Pi2 has no Bluetooth — PL011 UART available directly, no overlay needed

# ---- Raspberry Pi 3B ----
[pi3]
kernel=kernel8-32.img
gpu_mem=128
screen_width=800
screen_height=600
# Disable Bluetooth to free PL011 UART for serial debug on GPIO 14/15
dtoverlay=disable-bt
# Force HDMI audio
hdmi_drive=2
hdmi_force_edid_audio=1

# ---- Raspberry Pi 4B / 400 ----
[pi4]
kernel=kernel7l.img
gpu_mem=128
screen_width=800
screen_height=600
# Disable Bluetooth to free PL011 UART for serial debug on GPIO 14/15
dtoverlay=disable-bt
# Audio configuration - PWM works on all setups, HDMI requires compatible monitor
# sounddev=sndpwm          # PWM audio (headphone jack) - default, always works
# sounddev=sndhdmi         # HDMI audio (requires HDMI monitor with audio support)
sounddev=sndpwm

# ---- Back to all models ----
[all]
